    <!-- JavaScript Logika Aplikasi -->
    <script>
        // DOM Elements
        const form = document.getElementById('task-form');
        const taskInput = document.getElementById('task-input');
        const categoryInput = document.getElementById('category-input');
        const timeInput = document.getElementById('time-input');
        const taskList = document.getElementById('task-list');
        const emptyState = document.getElementById('empty-state');
        const progressText = document.getElementById('progress-text');

        // State: Mengambil data dari localStorage atau array kosong
        let tasks = JSON.parse(localStorage.getItem('noyz_schedule')) || [];

        // Mapping warna badge untuk setiap kategori
        const categoryColors = {
            'Rutinitas': 'bg-blue-100 text-blue-600',
            'Hobi': 'bg-purple-100 text-purple-600',
            'Belajar': 'bg-green-100 text-green-600',
            'Penting': 'bg-red-100 text-red-600',
            'Olahraga': 'bg-orange-100 text-orange-600'
        };

        // Fungsi menyimpan ke LocalStorage
        function saveTasks() {
            localStorage.setItem('noyz_schedule', JSON.stringify(tasks));
            renderTasks();
        }

        // Efek Konfeti! 🎉
        function fireConfetti() {
            confetti({
                particleCount: 100,
                spread: 70,
                origin: { y: 0.6 },
                colors: ['#ffb3ba', '#ffdfba', '#ffffba', '#baffc9', '#bae1ff', '#d4c4fb']
            });
        }

        // Render Jadwal ke Layar
        function renderTasks() {
            taskList.innerHTML = '';
            
            // Urutkan jadwal berdasarkan jam (dari yang paling pagi)
            tasks.sort((a, b) => a.time.localeCompare(b.time));

            const totalTasks = tasks.length;
            const completedTasks = tasks.filter(t => t.completed).length;

            // Update Progress Text
            progressText.innerText = `${completedTasks}/${totalTasks} Selesai`;
            if (totalTasks > 0 && completedTasks === totalTasks) {
                progressText.classList.replace('text-pink-500', 'text-green-500');
                progressText.classList.replace('bg-pink-100', 'bg-green-100');
                progressText.innerText = '✨ Semua Selesai! ✨';
            } else {
                progressText.classList.replace('text-green-500', 'text-pink-500');
                progressText.classList.replace('bg-green-100', 'bg-pink-100');
            }

            // Tampilkan atau sembunyikan gambar kosong
            if (totalTasks === 0) {
                emptyState.classList.remove('hidden');
                emptyState.classList.add('flex');
            } else {
                emptyState.classList.add('hidden');
                emptyState.classList.remove('flex');
            }

            // Buat elemen HTML untuk tiap jadwal
            tasks.forEach((task, index) => {
                const li = document.createElement('li');
                li.className = `task-enter bg-white p-4 rounded-2xl shadow-sm border border-gray-100 flex items-center gap-4 transition-all duration-300 hover:shadow-md ${task.completed ? 'opacity-60 bg-gray-50' : ''}`;
                li.style.animationDelay = `${index * 0.1}s`;

                li.innerHTML = `
                    <input type="checkbox" class="custom-checkbox flex-shrink-0" ${task.completed ? 'checked' : ''} onchange="toggleTask(${task.id})">
                    
                    <div class="flex-1 min-w-0">
                        <h3 class="font-bold text-gray-800 text-lg truncate ${task.completed ? 'line-through text-gray-400' : ''}">
                            ${task.text}
                        </h3>
                        <div class="flex items-center gap-2 mt-1">
                            <span class="text-xs font-bold px-2 py-0.5 rounded-md flex items-center gap-1 bg-gray-100 text-gray-600">
                                <i class="ph ph-clock"></i> ${task.time}
                            </span>
                            <span class="text-xs font-bold px-2 py-0.5 rounded-md ${categoryColors[task.category]}">
                                ${task.category}
                            </span>
                        </div>
                    </div>

                    <button onclick="deleteTask(${task.id})" class="w-9 h-9 rounded-full bg-red-50 text-red-400 hover:bg-red-100 hover:text-red-600 flex items-center justify-center transition-colors flex-shrink-0">
                        <i class="ph ph-trash text-lg"></i>
                    </button>
                `;
                taskList.appendChild(li);
            });
        }

        // Tambah Jadwal Baru
        form.addEventListener('submit', (e) => {
            e.preventDefault();
            
            const text = taskInput.value.trim();
            const category = categoryInput.value;
            const time = timeInput.value;

            if (text && time) {
                const newTask = {
                    id: Date.now(), // Unique ID
                    text: text,
                    category: category,
                    time: time,
                    completed: false
                };

                tasks.push(newTask);
                saveTasks();
                
                // Reset input (waktu dibiarkan agar mudah input jadwal berurutan)
                taskInput.value = '';
                taskInput.focus();
            }
        });

        // Toggle Centang Jadwal (Selesai/Belum)
        window.toggleTask = function(id) {
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.completed = !task.completed;
                
                // Jika baru saja dicentang menjadi selesai, keluarkan konfeti!
                if (task.completed) {
                    fireConfetti();
                }
                
                saveTasks();
            }
        };

        // Hapus Jadwal
        window.deleteTask = function(id) {
            // Animasi sedikit sebelum hapus (opsional, via konfirmasi)
            if(confirm("Yakin mau hapus jadwal ini? 🥺")) {
                tasks = tasks.filter(t => t.id !== id);
                saveTasks();
            }
        };

        // Render awal saat halaman dimuat
        renderTasks();
    </script>
